# Permite a jugadores usar skins de Minecraft premium

# Comandos:
# /skin set <nombre> - Cambiar skin
# /skin clear - Remover skin personalizada

# Este directorio contendrá:
# - config.yml (configuración principal)
# - Skins/ (caché de skins descargadas)
# - Players.yml (skins asignadas a jugadores)
